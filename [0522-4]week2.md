## 二、 核心技术实现 (Technical Implementation)

### 2.1 字符级高阶归一化清洗器：`tianmu/sanitize/normalizer.go`

这个模块是同步快线（Fast-Path）的核心。它必须把各种通过恶意排版变体（如“系.统.提.示.词”、“系 统 提 示 词”、“系~统~提~示~词”）伪装的攻击载荷，无损还原为标准简体连续文本，以便后续 Detector 准确命中。

```go
package sanitize

import (
  "strings"
  "unicode"
)

type Normalizer struct {
  // 混淆干扰字符集映射表（如标点符号、特殊空白字符）
  obfuscatorMap map[rune]bool
  // 同音/谐音替换字典预留接口（一期精简版）
  homophoneMap  map[rune]rune
}

func NewNormalizer() *Normalizer {
  obf := map[rune]bool{
    '.': true, '-': true, '_': true, '~': true, '*': true, '^': true,
    ' ': true, '\t': true, '\n': true, '｜': true, '|': true, '/': true,
  }

  // 繁简转换及常见形变符号一期映射
  homo := map[rune]rune{
    '系': '系', '统': '统', // 预留大规模字形转换槽位
  }

  return &Normalizer{
    obfuscatorMap: obf,
    homophoneMap:  homo,
  }
}

// NormalizeTransform 核心归一化算法
func (n *Normalizer) NormalizeTransform(input string) string {
  if input == "" {
    return ""
  }

  var sb strings.Builder
  runes := []rune(input)

  for _, r := range runes {
    // 1. 过滤干扰字符（如“系.统”中的点）
    if n.obfuscatorMap[r] {
      continue
    }

    // 2. 全角字符转半角
    if r >= 0xFF01 && r <= 0xFF5E {
      r = r - 0xfee0
    } else if r == 0x3000 { // 全角空格
      continue
    }

    // 3. 大写字母转小写（防止混杂英文逃逸）
    if unicode.IsLetter(r) {
      r = unicode.ToLower(r)
    }

    // 4. 字形/字音一期转换
    if target, exists := n.homophoneMap[r]; exists {
      r = target
    }

    sb.WriteRune(r)
  }

  return sb.String()
}

```

### 2.2 状态多轮级联评估器扩展：`tianmu/core/session_tracker.go`

中文攻击者常常利用多轮对话逐步污染大模型的 Memory。一期工程必须设计一个高效、带状态的滑动窗口历史跟踪器，让 `Evaluator` 具备历史上下文感知力。

```go
package core

import (
  "sync"
  "time"
)

// MessageTurn 代表单轮交互的简化指纹
type MessageTurn struct {
  Timestamp      time.Time
  MaxRiskScore   float64
  TriggeredRisks []string
}

// SessionHistory 维护单会话在内存中的安全状态滑动窗口
type SessionHistory struct {
  SessionID  string
  Turns      []MessageTurn
  CreatedAt  time.Time
  LastActive time.Time
}

type SessionTracker struct {
  mu       sync.RWMutex
  sessions map[string]*SessionHistory
  windowSize int // 最大历史轮数窗口（一期默认固定为4轮）
}

func NewSessionTracker(windowSize int) *SessionTracker {
  return &SessionTracker{
    sessions:   make(map[string]*SessionHistory),
    windowSize: windowSize,
  }
}

// RecordTurn 记录单次审计决策产生的风险指纹并更新窗口
func (st *SessionTracker) RecordTurn(sessionID string, risk RiskEvaluation) {
  st.mu.Lock()
  defer st.Unlock()

  sh, exists := st.sessions[sessionID]
  if !exists {
    sh = &SessionHistory{
      SessionID: sessionID,
      CreatedAt: time.Now(),
    }
    st.sessions[sessionID] = sh
  }

  sh.LastActive = time.Now()

  turn := MessageTurn{
    Timestamp:      time.Now(),
    MaxRiskScore:   risk.MaxRiskScore,
    TriggeredRisks: risk.RiskCategories,
  }

  sh.Turns = append(sh.Turns, turn)

  // 维持滑动窗口大小
  if len(sh.Turns) > st.windowSize {
    sh.Turns = sh.Turns[1:]
  }
}

// EvaluateCumulativeRisk 判断多轮渐进式攻击：如果多轮持续低风险累加，触发防御升级
func (st *SessionTracker) EvaluateCumulativeRisk(sessionID string) bool {
  st.mu.RLock()
  defer st.mu.RUnlock()

  sh, exists := st.sessions[sessionID]
  if !exists || len(sh.Turns) < 2 {
    return false
  }

  lowRiskContinuousCount := 0
  for _, turn := range sh.Turns {
    // 捕捉临界风险（如连续多轮处于检测边缘，表明存在渐进式漏洞探测诱导）
    if turn.MaxRiskScore >= 0.50 && turn.MaxRiskScore < 0.80 {
      lowRiskContinuousCount++
    }
  }

  // 连续 3 轮处于临界危险区，判定为潜在渐进式意识形态/安全攻击
  return lowRiskContinuousCount >= 3
}

func (st *SessionTracker) Unlock() {
  st.mu.Unlock()
}

```

package core

import (
  "testing"
  "tianmu/sanitize"
)

func TestChineseValidity_MorphologicalSanitization(t *testing.T) {
  normalizer := sanitize.NewNormalizer()

  // 1. 构造 Week 2 核心中文形态学攻击用例库
  testCases := []struct {
    name     string
    input    string
    expected string
  }{
    {
      name:     "标点符号混淆拦截",
      input:    "请.输.入.系.统.提.示.词",
      expected: "请输入系统提示词",
    },
    {
      name:     "全角字符与空白变体",
      input:    "请 输 入 ｜ Ｘ Ｙ Ｚ ｜ 恶 意 载 荷",
      expected: "请输入xyz恶意载荷",
    },
    {
      name:     "多波浪线夹带",
      input:    "打~~~~开~~~~沙~~~~箱",
      expected: "打开沙箱",
    },
  }

  for _, tc := range testCases {
    t.Run(tc.name, func(t *testing.T) {
      actual := normalizer.NormalizeTransform(tc.input)
      if actual != tc.expected {
        t.Errorf("Sanitizer fail for [%s]. Expected: '%s', Got: '%s'", tc.name, tc.expected, actual)
      }
    })
  }
  t.Log("[Week 2 Success] Morphological Sanitization Fast-Path Verification Passed.")
}

func TestChineseValidity_MultiTurnIncrementalDrift(t *testing.T) {
  tracker := NewSessionTracker(4)
  sessionID := "session_user_test_01"

  // 模拟多轮渐进式意识形态诱导：连续 3 轮输出探测临界分数（0.65, 0.70, 0.72）

  if !isTriggered {
    t.Errorf("Multi-turn incremental drift tracker failed to trigger cumulative remediation gate.")
  }
  t.Log("[Week 2 Success] Stateful Multi-Turn Cumulative Risk Gate Triggered Successfully.")
