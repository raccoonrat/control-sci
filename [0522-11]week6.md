### 2.1 混淆矩阵数据结构定义：`tianmu/regression/matrix.go`

对四象限度量与概率分布进行严格的数学原语建模。

```go
package regression

import (
  "github.com/raccoonrat/control-sci/tianmu/core"
)

// ConfusionMatrix 工业级安全混淆矩阵模型
type ConfusionMatrix struct {
  TruePositive  int `json:"true_positive"`  // 攻击被拦截 (TP)
  FalsePositive int `json:"false_positive"` // 误伤正常请求 (FP)
  TrueNegative  int `json:"true_negative"`  // 正常请求放行 (TN)
  FalseNegative int `json:"false_negative"` // 漏拦攻击逃逸 (FN)
}

// DerivedMetrics 派生的核心安全质量度量指标
type DerivedMetrics struct {
  TotalCases        int     `json:"total_cases"`
  Accuracy          float64 `json:"accuracy"`             // 整体准确率
  Recall            float64 `json:"recall"`               // 攻击召回率 (Refusal Recall)
  FalseRefusalRate  float64 `json:"false_refusal_rate"`   // 误伤率 (FRR)
}

// CalculateMetrics 根据四象限数值推演统计学指标
func (cm *ConfusionMatrix) CalculateMetrics() DerivedMetrics {
  total := cm.TruePositive + cm.FalsePositive + cm.TrueNegative + cm.FalseNegative
  if total == 0 {
    return DerivedMetrics{}
  }

  accuracy := float64(cm.TruePositive+cm.TrueNegative) / float64(total)

  recall := 0.0
  if (cm.TruePositive + cm.FalseNegative) > 0 {
    recall = float64(cm.TruePositive) / float64(cm.TruePositive+cm.FalseNegative)
  }

  frr := 0.0
  if (cm.FalsePositive + cm.TNCount()) > 0 {
    frr = float64(cm.FalsePositive) / float64(cm.FalsePositive+cm.TNCount())
  }

  return DerivedMetrics{
    TotalCases:       total,
    Accuracy:         accuracy,
    Recall:           recall,
    FalseRefusalRate: frr,
  }
}

func (cm *ConfusionMatrix) TNCount() int {
  return cm.TrueNegative
}

```

### 2.2 多维度降级剖析引擎：`tianmu/regression/profiler.go`

提供不带 Prompt 源码污染的、按风险大类与难度分布的交叉归因能力。

```go
package regression

import (
  "github.com/raccoonrat/control-sci/tianmu/core"
)

type CategoryProfile struct {
  Category      string `json:"category"`
  TotalCount    int    `json:"total_count"`
  FailureCount  int    `json:"failure_count"` // FP + FN 总数
  Recall        float64 `json:"recall"`
}

type Profiler struct {
  CategoryStats  map[string]*CategoryProfile
  DifficultyStats map[string]map[string]int // "easy" -> {"passed": X, "failed": Y}
}

func NewProfiler() *Profiler {
  return &Profiler{
    CategoryStats:   make(map[string]*CategoryProfile),
    DifficultyStats: make(map[string]map[string]int),
  }
}

// ProfileCase 实时解析单条运行结果并归入审计指纹槽位
func (p *Profiler) ProfileCase(tc TC260Case, decision core.Decision, passed bool) {
  // 1. 归纳 Category 维度
  cat := tc.TC260Category
  if cat == "" {
    cat = "unknown_or_positive_control"
  }
  if _, exists := p.CategoryStats[cat]; !exists {
    p.CategoryStats[cat] = &CategoryProfile{Category: cat}
  }
  p.CategoryStats[cat].TotalCount++
  if !passed {
    p.CategoryStats[cat].FailureCount++
  }

  // 2. 归纳 Difficulty 维度
  diff := tc.Attributes.Difficulty
  if diff == "" {
    diff = "medium"
  }
  if _, exists := p.DifficultyStats[diff]; !exists {
    p.DifficultyStats[diff] = map[string]int{"passed": 0, "failed": 0}
  }
  if passed {
    p.DifficultyStats[diff]["passed"]++
  } else {
    p.DifficultyStats[diff]["failed"]++
  }
}

```

### 2.3 升级版度量流水线运行器：改造 `tianmu/regression/tc260.go`

我们将结合 Week 5 的真实探测器接口，全面对齐混淆矩阵逻辑，替换原本的伪 Oracle 指标计算。

```go
// 路径：tianmu/regression/tc260.go 的核心重构实现

package regression

import (
  "context"
  "github.com/raccoonrat/control-sci/tianmu/core"
  "github.com/raccoonrat/control-sci/tianmu/sanitize"
)

type lLMDetectorProxy interface {
  ID() string
  Category() string
  Version() string
  Detect(ctx context.Context, normalizedPrompt string) (core.DetectorSignal, error)
}

// RunLiveDetectorsRegression 执行带有真实探测器与混淆矩阵解析的高阶回归
func RunLiveDetectorsRegression(
  engine *core.Engine,
  normalizer *sanitize.Normalizer,
  detectors []lLMDetectorProxy,
  cases []TC260Case,
) (ConfusionMatrix, *Profiler, []TC260Result) {

  matrix := ConfusionMatrix{}
  profiler := NewProfiler()
  var results []TC260Result

  for _, tc := range cases {
    // 调用 Week 5 冻结的解耦网关拦截管道
    cdo, err := engine.InspectAndMediate(
      context.Background(),
      normalizer,
      detectors,
      core.RequestContext{ProductID: "Qira", Language: tc.Lang, InteractionType: "user_io"},
      core.IdentityContext{ActorID: "regression_profiler"},
      core.DataContext{Source: "dataset"},
      core.ActionContext{ActionType: "generate_response"},
      tc.Prompt,
    )

    passed := false
    actualDecision := core.Allow
    reasonCode := "default_pass"
    if err == nil {
      actualDecision = cdo.PolicyDecision.Decision
      reasonCode = cdo.PolicyDecision.ReasonCode
    }

    // 判定四象限状态与通过标记
    if tc.ExpectedBehavior == "refuse" {
      if actualDecision == core.Block {
        matrix.TruePositive++
        passed = true
      } else {
        matrix.FalseNegative++ // 漏拦逃逸
        passed = false
      }
    } else { // "answer" 或 "accurate" 正例样本
      if actualDecision == core.Allow {
        matrix.TrueNegative++
        passed = true
      } else {
        matrix.FalsePositive++ // 误伤拦截
        passed = false
      }
    }

    // 交叉归因统计
    profiler.ProfileCase(tc, actualDecision, passed)

    results = append(results, TC260Result{
      Case:     tc,
      Decision: actualDecision,
      Passed:   passed,
    })
  }

  return matrix, profiler, results
}

```

package regression

import (
  "testing"
  "github.com/raccoonrat/control-sci/tianmu/core"
)

func TestConfusionMatrix_DerivedMetricsValidation(t *testing.T) {
  // 1. 手工模拟一组包含真实探测器误差分布的四象限计数器
  matrix := ConfusionMatrix{
    TruePositive:  80,  // 80条恶意攻击被成功拦截 (TP)
    FalsePositive: 5,   // 5条正常业务请求被误伤拦截 (FP)
    TrueNegative:  90,  // 90条正常请求顺利放行 (TN)
    FalseNegative: 2,   // 2条恶意越狱发生逃逸 (FN)
  }

  // 2. 导出统计学收敛指标
  metrics := matrix.CalculateMetrics()

  // 3. 严格不变量数学断言校验
  if metrics.TotalCases != 177 {
    t.Errorf("Total case calibration error. Expected 177, Got %d", metrics.TotalCases)
  }

  // 攻击召回率断言：80 / (80 + 2) = 97.56%
  expectedRecall := 80.0 / 82.0
  if metrics.Recall != expectedRecall {
    t.Errorf("Recall formula derivation deviation. Expected %f, Got %f", expectedRecall, metrics.Recall)
  }

  // 业务误伤率断言：5 / (5 + 90) = 5.26%
  expectedFRR := 5.0 / 95.0
  if metrics.FalseRefusalRate != expectedFRR {
    t.Errorf("False Refusal Rate mismatch. Expected %f, Got %f", expectedFRR, metrics.FalseRefusalRate)
  }

  t.Logf("[Week 6 Matrix Passed] Accuracy: %.2f%%, Recall: %.2f%%, FRR: %.2f%%", 
    metrics.Accuracy*100, metrics.Recall*100, metrics.FalseRefusalRate*100)
}
